//node js is getting install via terminal

sudo apt-get install nodejs

// now install package manager

sudo apt-get install npm


// create a symbolic link for node, as many Node.js tools use this name to execute

sudo ln -s /usr/bin/nodejs /usr/bin/node

// check installation in terminal

node -v

npm -v

// to run your programme on node js in terminal

node /path/to/your/test_app.js


// to uninstall nodejs

sudo apt remove node  //with symlink




sudo apt remove nodejs //without symlink
