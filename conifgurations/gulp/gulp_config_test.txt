// Create a new gulpfile.js configuration file in the root

// Gulp.js configuration
  var
    // modules
    gulp = require('gulp'),

    // development mode?
    devBuild = (process.env.NODE_ENV !== 'production'),

    // folders
    folder = {
      src: 'src/',
      build: 'build/'
    }
  ;
//

//now lets have an example with some gulp packages
//lets install package

$ npm install gulp-htmlclean --save-dev


// this is then referenced at the top of gulpfile.js

var
 // modules
 gulp = require('gulp'),
 htmlclean = require('gulp-htmlclean'),

//We can now create an html task at the end of gulpfile.js:


  // HTML processing
  gulp.task('html', ['images'], function() {
    var
      out = folder.build + 'html/',
      page = gulp.src(folder.src + 'html/**/*');

    // minify production code
    if (!devBuild) {
      page = page.pipe(htmlclean());
    }

    return page.pipe(gulp.dest(out));
  });



// now lets create JAvaScript task

 $ npm install gulp-concat gulp-uglify --save-dev

// Reference them at the top of gulpfile.js:


var
  ...
  concat = require('gulp-concat'),
  uglify = require('gulp-uglify'),


// Then add a new js task:

// JavaScript processing
gulp.task('js', function() {

  var jsbuild = gulp.src(folder.src + 'js/**/*')

    .pipe(concat('main.js'));

  if (!devBuild) {
    jsbuild = jsbuild

      .pipe(uglify());
  }

  return jsbuild.pipe(gulp.dest(folder.build + 'js/'));

});

// now run in the project directory  :

$ gulp js
